Modal component

```jsx react-live=true noInline
render(() => {
  const [{ animation, state }, toggleFade] = useFade({ startOnRender: false });

  return (
    <>
      <Button onClick={() => toggleFade(true)}>Open modal</Button>
      {state && (
        <Modal
          animation={animation}
          onClose={() => toggleFade(false)}
          closeOnClickOutside={false}
        >
          <a href="#">link</a>
          <button>button</button>
          <input />
        </Modal>
      )}
    </>
  );
});
```
